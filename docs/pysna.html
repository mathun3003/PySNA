 145    def _manual_request(self, url: str, additional_fields: Dict[str, List[str]] | None = None) -> dict:
 146        """Perform a manual GET request to the Twitter API.
 147
 148        Args:
 149            url (str): API URL (without specified fields)
 150            additional_fields (Dict[str, List[str]] | None, optional): Fields can be specified (e.g., tweet.fields) according to the official API reference. Defaults to None.
 151
 152        Raises:
 153            Exception: If status code != 200.
 154
 155        Returns:
 156            dict: JSON formatted response of API request.
 157
 158        Reference: https://developer.twitter.com/en/docs/twitter-api/tweets/lookup/api-reference/get-tweets-id
 159        """
 160        # if additional_fields were provided
 161        if additional_fields:
 162            # init empty string
 163            fields = "?"
 164            # create fields string dynamically for every field in additional_fields
 165            for field in additional_fields.keys():
 166                # e.g., in format "tweet.fields=lang,author_id"
 167                fields += f"{field}={','.join(additional_fields[field])}&"
 168            # append fields to url
 169            url += fields[:-1]
 170        # set header
 171        header = {"Authorization": f"Bearer {self._bearer_token}"}
 172        response = requests.get(url, headers=header)
 173        if response.status_code != 200:
 174            raise Exception("Request returned an error: {} {}".format(response.status_code, response.text))
 175        return response.json()

 321    def _calc_datetime_metrics(self, data: Dict[str, datetime]) -> dict():
 322        # use the datetime's timestamp to make them comparable
 323        timestamps = [dt.timestamp() for dt in data.values()]
 324        # calc mean of creation dates
 325        total_time = sum(timestamps)
 326        mean_timestamp = total_time / len(timestamps)
 327        # convert mean timestamp back to datetime object with timezone information
 328        mean_datetime = datetime.fromtimestamp(mean_timestamp, tz=timezone.utc)
 329
 330        # calculate time differences to mean datetime of every creation date
 331        time_diffs_mean = {user: {"days": (dt - mean_datetime).days, "seconds": (dt - mean_datetime).seconds} for user, dt in data.items()}
 332
 333        # find the median of the timestamps
 334        median_timestamp = np.median(timestamps)
 335        # Convert median timestamp back to datetime object
 336        median_datetime = datetime.fromtimestamp(median_timestamp, tz=timezone.utc)
 337
 338        # calculate time differences to median timestamp of every creation date
 339        time_diffs_median = {user: {"days": (dt - median_datetime).days, "seconds": (dt - median_datetime).seconds} for user, dt in data.items()}
 340
 341        # calc range of creation dates
 342        max_date, min_date = max(data.values()), min(data.values())
 343        time_span = max_date - min_date
 344
 345        # convert creation dates to isoformat for readability
 346        data = {u: dt.isoformat() for u, dt in data.items()}
 347
 348        # add metrics to output
 349        data["metrics"] = dict()
 350        data["metrics"]["deviation_from_mean"] = time_diffs_mean
 351        data["metrics"]["deviation_from_median"] = time_diffs_median
 352        data["metrics"]["time_span"] = {"days": time_span.days, "seconds": time_span.seconds, "microseconds": time_span.microseconds}
 353        data["metrics"]["mean"] = mean_datetime.isoformat()
 354        data["metrics"]["median"] = median_datetime.isoformat()
 355        data["metrics"]["max"] = max_date.isoformat()
 356        data["metrics"]["min"] = min_date.isoformat()
 357        return data
